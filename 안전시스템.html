export default function App() {

  // 인증 및 라우팅 상태

  const [appMode, setAppMode] = useState('login'); // 'login', 'admin', 'system'

  // 관리자 로그인 및 설정용 상태 추가

  const [adminIdInput, setAdminIdInput] = useState('');

  const [adminPwInput, setAdminPwInput] = useState('');

  const [adminConfig, setAdminConfig] = useState(null);

  const [isSettingsModalOpen, setIsSettingsModalOpen] = useState(false);

  const [editAdminId, setEditAdminId] = useState('');

  const [editAdminPw, setEditAdminPw] = useState('');

  useEffect(() => {

    const initAuth = async () => {

      try {

        if (typeof __initial_auth_token !== 'undefined' && __initial_auth_token) {

          await signInWithCustomToken(auth, __initial_auth_token);

        } else {

          await signInAnonymously(auth);

        }

      } catch (error) {

        console.error("인증 실패:", error);

      }

    };

    initAuth();

    const unsubscribe = onAuthStateChanged(auth, setUser);

    return () => unsubscribe();

  }, []);


  // 2. 관리자용 전체 고객사 목록 실시간 구독

  useEffect(() => {

    if (!user) return;

    const compRef = collection(db, 'artifacts', appId, 'public', 'data', 'companies');

    const unsubscribe = onSnapshot(compRef, (snap) => {

      const compData = snap.docs.map(d => ({ id: d.id, ...d.data() }));

      setCompanies(compData);

    }, (error) => console.error("고객사 목록 구독 오류:", error));

    

    return () => unsubscribe();

  }, [user]);


  // 관리자 계정 정보 실시간 구독 및 초기화 (Firestore 동기화)

  useEffect(() => {

    if (!user) return;

    const configRef = doc(db, 'artifacts', appId, 'public', 'data', 'config', 'admin_settings');

    const unsubscribe = onSnapshot(configRef, (docSnap) => {

      if (docSnap.exists()) {

        setAdminConfig(docSnap.data());

      } else {

        // 최초 접속 시 기본 관리자 계정 생성

        const defaultConfig = { adminId: 'admin', adminPw: 'kpmc1234' };

        setDoc(configRef, defaultConfig);

        setAdminConfig(defaultConfig);

      }

    }, (error) => console.error("관리자 설정 구독 오류:", error));

    

    return () => unsubscribe();

  }, [user]);


  // 3. 특정 고객사의 ISO 시스템 데이터 실시간 구독

  useEffect(() => {

    if (!user || !currentCompany) return;

    

    const docRef = doc(db, 'artifacts', appId, 'public', 'data', 'company_data', currentCompany.id);

    const unsubscribe = onSnapshot(docRef, (docSnap) => {

      if (docSnap.exists()) {

        setSystemData(docSnap.data().systemData);

      } else {

        // 새 고객사의 경우 초기 데이터 세팅

        setSystemData(initialIsoData);

      }

    }, (error) => {

      console.error("Firestore 구독 오류:", error);

      showToast('고객사 데이터를 불러오는 중 오류가 발생했습니다.');

    });

    

    return () => unsubscribe();

  }, [user, currentCompany]);


  const showToast = (message) => {

    setToastMessage(message);

    setTimeout(() => setToastMessage(null), 3000);

  };


  // KPMC 관리자 로그인 처리 (Firestore 데이터 기반 검증)

  const handleAdminLogin = () => {

    if (!adminConfig) {

      showToast('서버 설정을 불러오는 중입니다. 잠시 후 다시 시도해주세요.');

      return;

    }

    if (adminIdInput === adminConfig.adminId && adminPwInput === adminConfig.adminPw) {

      setIsAdmin(true);

      setAppMode('admin');

      setAdminIdInput('');

      setAdminPwInput('');

      showToast('관리자 포털에 접속했습니다.');

    } else {

      showToast('아이디 또는 비밀번호가 일치하지 않습니다.');

    }

  };


  // KPMC 관리자 계정 정보 변경

  const handleUpdateAdminConfig = async () => {

    if (!editAdminId.trim() || !editAdminPw.trim()) {

      return showToast('아이디와 비밀번호를 모두 입력해주세요.');

    }

    try {

      const configRef = doc(db, 'artifacts', appId, 'public', 'data', 'config', 'admin_settings');

      await setDoc(configRef, { adminId: editAdminId, adminPw: editAdminPw });

      setIsSettingsModalOpen(false);

      showToast('관리자 계정 정보가 성공적으로 변경되었습니다.');

    } catch (e) {

      console.error(e);

      showToast('계정 정보 변경 중 오류가 발생했습니다.');

    }

  };


  // 고객사 접속 코드 로그인 처리

  const handleClientLogin = () => {

    if (!loginCode.trim()) {

      showToast('접속 코드를 입력해주세요.');

      return;

    }

    const comp = companies.find(c => c.code === loginCode.trim());

    if (comp) {

      setCurrentCompany(comp);

      setIsAdmin(false);

      setAppMode('system');

      showToast(`${comp.name} 시스템에 접속했습니다.`);

    } else {

      showToast('유효하지 않은 고객사 코드입니다.');

    }

  };


  // 로그아웃 처리

  const handleLogout = () => {

                </div>

                />

            </div>

          </main>

        )}


        {/* Dashboard View */}

        {viewMode === 'dashboard' && (

          <main className="flex-1 flex flex-col bg-slate-100 p-8 overflow-y-auto">

            <div className="max-w-6xl mx-auto w-full space-y-8">

              

              <div className="bg-white p-8 rounded-lg shadow-sm border-l-4 border-indigo-600">

                <h2 className="text-2xl font-bold text-slate-900 mb-2 tracking-tight">[{currentCompany?.name}] 문서화 진행 현황 대시보드</h2>

              </div>


              {/* Stat Cards */}

              <div className="grid grid-cols-1 md:grid-cols-3 gap-6">

                <div className="bg-white p-6 rounded-lg border border-slate-200 shadow-sm flex items-center gap-5">

                  <div className="bg-slate-100 p-4 rounded-full"><FileText className="w-7 h-7 text-slate-700" /></div>

                  <div>

                    <p className="text-sm text-slate-500 font-bold uppercase tracking-wider mb-1">총 요구 문서</p>

                    <p className="text-3xl font-black text-slate-900">{totalDocs}<span className="text-lg text-slate-400 font-medium ml-1">건</span></p>

                    <p className="text-sm text-slate-500 font-bold uppercase tracking-wider mb-1">승인/완료</p>

                    <p className="text-3xl font-black text-slate-900">{completedDocs}<span className="text-lg text-slate-400 font-medium ml-1">건</span></p>

                  <div className="bg-amber-50 p-4 rounded-full"><Clock className="w-7 h-7 text-amber-600" /></div>

                              </div>

                            </div>

                          ))}

                        </div>

                      </div>

                    )

                  })}

                </div>

              </div>


            </div>

          </main>

        )}

      </div>


      {/* Toast Notification */}

      {toastMessage && (

          <span className="text-sm font-medium tracking-wide">{toastMessage}</span>

        </div>

      )}

      {isAddModalOpen && (

        <div className="fixed inset-0 bg-slate-900/50 backdrop-blur-sm flex items-center justify-center z-50">

          <div className="bg-white rounded-xl shadow-2xl w-full max-w-md overflow-hidden animate-fade-in-up">

              </button>

            <div className="p-6 space-y-4">

              <div>

                <label className="block text-sm font-semibold text-slate-700 mb-1">문서 번호 / 식별자 <span className="text-red-500">*</span></label>
